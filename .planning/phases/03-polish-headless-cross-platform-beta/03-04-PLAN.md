Create `.github/workflows/ci-matrix.yml` that replaces `windows-smoke.yml` scope with a full cross-platform matrix:

```yaml
name: ci-matrix
on:
  push:
    branches: [master]
  pull_request:
    branches: [master]
permissions:
  contents: read
jobs:
  test:
    strategy:
      fail-fast: false
      matrix:
        os: [ubuntu-latest, windows-latest, macos-latest]
    runs-on: ${{ matrix.os }}
    timeout-minutes: 20
    steps:
      - uses: actions/checkout@v4
      - uses: oven-sh/setup-bun@v2
        with:
          bun-version: ">=1.3.13"
      - name: Install
        run: bun install --frozen-lockfile
      - name: Typecheck
        run: bunx tsc --noEmit
      - name: Test
        run: bunx vitest run
      - name: Boot smoke (headless)
        shell: bash
        run: |
          bun run src/index.ts --smoke-boot-only
          echo "Boot smoke OK (exit 0)"
  build-smoke:
    strategy:
      fail-fast: false
      matrix:
        include:
          - os: ubuntu-latest
            target: bun-linux-x64
            outfile: dist/muonroi-cli-linux-x64
          - os: windows-latest
            target: bun-windows-x64
            outfile: dist/muonroi-cli-windows-x64.exe
          - os: macos-latest
            target: bun-darwin-arm64
            outfile: dist/muonroi-cli-darwin-arm64
    runs-on: ${{ matrix.os }}
    timeout-minutes: 15
    steps:
      - uses: actions/checkout@v4
      - uses: oven-sh/setup-bun@v2
        with:
          bun-version: ">=1.3.13"
      - name: Install
        run: bun install --frozen-lockfile
      - name: Build binary
        run: bun build --compile --target=${{ matrix.target }} ./src/index.ts --outfile ${{ matrix.outfile }}
      - name: Verify binary exists
        shell: bash
        run: |
          ls -la ${{ matrix.outfile }}*
          echo "Binary compiled successfully for ${{ matrix.target }}"
```

Note: The `build-smoke` job only verifies the binary compiles — it does NOT attempt to run it (cross-compile on macos-latest for arm64 cannot run on x64 runner). Running the binary is verified on matching platform in the `test` job via `--smoke-boot-only`.

**Release workflow (.github/workflows/release-binary.yml):**

```yaml
name: release-binary
on:
  push:
    tags:
      - "v*"
permissions:
  contents: write
jobs:
  build:
    strategy:
      fail-fast: false
      matrix:
        include:
          - os: ubuntu-latest
            target: bun-linux-x64
            outfile: muonroi-cli-linux-x64
          - os: windows-latest
            target: bun-windows-x64
            outfile: muonroi-cli-windows-x64.exe
          - os: macos-latest
            target: bun-darwin-x64
            outfile: muonroi-cli-darwin-x64
          - os: macos-latest
            target: bun-darwin-arm64
            outfile: muonroi-cli-darwin-arm64
    runs-on: ${{ matrix.os }}
    timeout-minutes: 20
    steps:
      - uses: actions/checkout@v4
      - uses: oven-sh/setup-bun@v2
        with:
          bun-version: ">=1.3.13"
      - name: Install
        run: bun install --frozen-lockfile
      - name: Build binary
        run: bun build --compile --target=${{ matrix.target }} ./src/index.ts --outfile dist/${{ matrix.outfile }}
      - name: Upload artifact
        uses: actions/upload-artifact@v4
        with:
          name: ${{ matrix.outfile }}
          path: dist/${{ matrix.outfile }}
          retention-days: 5

  release:
    needs: build
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Download all artifacts
        uses: actions/download-artifact@v4
        with:
          path: dist/
          merge-multiple: true
      - name: Create GitHub Release
        env:
          GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}
        run: |
          gh release create ${{ github.ref_name }} \
            --title "muonroi-cli ${{ github.ref_name }}" \
            --notes "## muonroi-cli ${{ github.ref_name }}

          ### Installation

          **npm:** \`npm install -g muonroi-cli\`

          **Standalone binary:** Download for your platform below.

          > **Note:** Standalone binaries use env-var API key (\`ANTHROPIC_API_KEY\`). OS keychain integration requires npm install.
          " \
            dist/*

  npm-publish:
    needs: build
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: oven-sh/setup-bun@v2
        with:
          bun-version: ">=1.3.13"
      - name: Install
        run: bun install --frozen-lockfile
      - name: Publish to npm
        env:
          NPM_TOKEN: ${{ secrets.NPM_TOKEN }}
        run: |
          echo "//registry.npmjs.org/:_authToken=${NPM_TOKEN}" > .npmrc
          npm publish --access public
```

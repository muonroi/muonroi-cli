```typescript
// CQ-14: Auto-add Experience Auditor stance when EE returns >= 1 high-confidence warning
if (eeWarnings && eeWarnings.length >= 1 && experienceMode !== "off") {
  const topWarning = eeWarnings[0];
  const auditorStance: DebateStance = {
    name: "Experience Auditor",
    lens: "Challenge all claims against known past mistakes and principles recorded in the experience brain.",
    focus: eeWarnings.map((w) => w.text).join("; ").slice(0, 300),
  };
  // advisory: append as 3rd voice; enforcing: replace last generic stance
  if (experienceMode === "enforcing" && plan.stances.length >= 2) {
    plan.stances[plan.stances.length - 1] = auditorStance;
  } else {
    plan.stances.push(auditorStance);
  }
  void topWarning; // suppress unused var lint
}
```

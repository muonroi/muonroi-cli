      ```typescript
      // src/orchestrator/pending-calls.ts
      import { promises as fs } from "node:fs";
      import * as path from "node:path";
      import * as crypto from "node:crypto";

      export interface PendingCallEntry {
        call_id: string;
        tool_name: string;
        started_ms: number;
        ended_ms?: number;
        status: "pending" | "settled" | "aborted" | "abandoned";
        staged_paths?: string[];
      }

      export function stableCallId(turnId: string, toolName: string, input: unknown): string {
        const h = crypto.createHash("sha256");
        h.update(turnId);
        h.update(":");
        h.update(toolName);
        h.update(":");
        h.update(JSON.stringify(input ?? null));
        return h.digest("hex").slice(0, 16);
      }

      export interface PendingCallsLog {
        begin(entry: Omit<PendingCallEntry, "started_ms" | "status"> & { staged_paths?: string[] }): Promise<void>;
        end(call_id: string, status: "settled" | "aborted"): Promise<void>;
        reconcile(): Promise<{ abandoned: number; settled: number }>;
      }

      export function createPendingCallsLog(sessionDir: string): PendingCallsLog {
        const filePath = path.join(sessionDir, "pending_calls.jsonl");
        let writeChain: Promise<void> = Promise.resolve();

        async function appendLine(obj: unknown): Promise<void> {
          await fs.mkdir(path.dirname(filePath), { recursive: true });
          await fs.appendFile(filePath, JSON.stringify(obj) + "\n", "utf8");
        }
        function serialize(work: () => Promise<void>): Promise<void> {
          writeChain = writeChain.then(work, work);
          return writeChain;
        }

        async function readEntries(): Promise<PendingCallEntry[]> {
          try {
            const raw = await fs.readFile(filePath, "utf8");
            return raw.split("\n").filter(Boolean).map(line => JSON.parse(line) as PendingCallEntry);
          } catch (err) {
            if ((err as NodeJS.ErrnoException).code === "ENOENT") return [];
            throw err;
          }
        }

        return {
          async begin(entry) {
            const full: PendingCallEntry = {
              ...entry,
              started_ms: Date.now(),
              status: "pending",
            };
            await serialize(() => appendLine({ event: "begin", ...full }));
          },
          async end(call_id, status) {
            await serialize(() => appendLine({ event: "end", call_id, status, ended_ms: Date.now() }));
          },
          async reconcile() {
            const lines = await readEntries();
            // Latest-status-per-call_id: walk and apply events to a map
            const byId = new Map<string, PendingCallEntry>();
            for (const line of lines as Array<PendingCallEntry & { event?: string }>) {
              if ((line as any).event === "begin") {
                byId.set(line.call_id, { ...line, status: "pending" });
              } else if ((line as any).event === "end") {
                const existing = byId.get(line.call_id);
                if (existing) byId.set(line.call_id, { ...existing, status: line.status as any, ended_ms: line.ended_ms });
              }
            }
            const stillPending = [...byId.values()].filter(e => e.status === "pending");
            for (const entry of stillPending) {
              for (const tmp of entry.staged_paths ?? []) {
                if (!tmp.endsWith(".tmp")) continue; // safety: only act on .tmp paths
                const final = tmp.slice(0, -4);
                try {
                  const tmpExists = await fs.access(tmp).then(() => true).catch(() => false);
                  const finalExists = await fs.access(final).then(() => true).catch(() => false);
                  if (tmpExists && !finalExists) {
                    // The atomic rename never happened — best-effort: unlink the .tmp (rollback).
                    await fs.unlink(tmp);
                  } else if (tmpExists && finalExists) {
                    // Rename completed pre-crash but the cleanup didn't — unlink the orphan .tmp.
                    await fs.unlink(tmp);
                  }
                } catch (err) {
                  console.warn(`[muonroi-cli] reconcile: could not clean ${tmp}: ${(err as Error).message}`);
                }
              }
              await serialize(() => appendLine({ event: "end", call_id: entry.call_id, status: "abandoned", ended_ms: Date.now() }));
            }
            return { abandoned: stillPending.length, settled: 0 };
          },
        };
      }
      ```

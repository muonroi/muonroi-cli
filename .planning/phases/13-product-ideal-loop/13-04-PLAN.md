```ts
// CB-1
recent = history.slice(-3).map(s => s.actualCost);
ewma = recent.reduce((avg, c) => avg * 0.7 + c * 0.3, recent[0]);
projection = ewma * 1.2;
halt = projection > (capUsd - spentUsd) * 1.5;

// CB-2
halt = sprintN >= 3 && delta_t <= 0 && delta_t_minus_1 <= 0;
// where delta_t = sprint[t].score - sprint[t-1].score

// CB-3
halt = sprintN === 1 && (recipe === null || recipe.coverage === 0);

// Cond #2 evidence regex (5 forms exactly)
/\b\w+\.(ts|tsx|js|py|go|rs|java):\d+/
/\btest\(['"`].+['"`]\)|describe\(['"`].+['"`]\)/
/\b[a-f0-9]{7,40}\b/
/\b(?:lighthouse|p95|p99|qps|throughput)[\s:=]+\d+/i
/\b(GET|POST|PUT|DELETE|PATCH)\s+\/[^\s]+\s*→\s*\d{3}\b/

// Cond #3 weighted score
score = sum(weight * statusValue) / sum(weight)
statusValue = met:1 | partial:0.5 | unmet:0
threshold default 0.9, range [0.7, 1.0], hard floor 0.7

// Cond #4 debate rounds matrix
sameProvider, sameModel       → REFUSE (echo_chamber)
crossProvider                 → 1 round
sameProvider, differentTier   → 3 rounds
sameProvider, sameTier, diffM → 5 rounds + explicit dissent
score < 0.85                  → SKIP Cond #4 entirely (R5)
```

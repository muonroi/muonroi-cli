        { re: /\b(search|find|grep|look\s+for)\b/i, intent: 'search', confidence: 0.80, modelHint: 'claude-3-5-haiku-latest' },

    Step 2 — src/router/classifier/tree-sitter.ts:
      ```typescript
      import type { ClassifierResult } from '../types.js';
      import { loadGrammarBytes, type GrammarId } from './grammars.js';

      type ParserHandle = { parser: any; lang: any };
      const cache = new Map<GrammarId, ParserHandle>();
      let initOnce: Promise<void> | null = null;

      async function initParser(): Promise<void> {
        if (initOnce) return initOnce;
        initOnce = (async () => {
          const ts = await import('web-tree-sitter');
          await ts.Parser.init();
        })();
        return initOnce;
      }

      export async function initTreeSitter(grammars: GrammarId[] = ['typescript','python']): Promise<void> {
        await initParser();
        const ts = await import('web-tree-sitter');
        for (const id of grammars) {
          if (cache.has(id)) continue;
          const bytes = await loadGrammarBytes(id);
          const lang = await ts.Language.load(bytes);
          const parser = new ts.Parser();
          parser.setLanguage(lang);
          cache.set(id, { parser, lang });
        }
      }
      // Fire-and-forget warm at boot
      export function warmTreeSitter(): void { void initTreeSitter(); }

      function detectLang(prompt: string): GrammarId | null {
        if (/```(ts|tsx|typescript)\b/i.test(prompt)) return 'typescript';
        if (/```(py|python)\b/i.test(prompt)) return 'python';
        return null;
      }

      export function lazyTreeSitter(prompt: string): ClassifierResult {
        const id = detectLang(prompt);
        if (!id) return { tier: 'abstain', confidence: 0.0, reason: 'tree-sitter:no-fenced-code' };
        const handle = cache.get(id);
        if (!handle) return { tier: 'abstain', confidence: 0.30, reason: 'tree-sitter:cold' };
        // Extract fenced code body (best-effort)
        const m = prompt.match(/```\w+\n([\s\S]*?)```/);
        const body = m ? m[1] : prompt;
        try {
          const tree = handle.parser.parse(body);
          // Confidence based on syntactically meaningful node count
          const root = tree.rootNode;
          const named = root.namedChildCount;
          const errs = root.hasError ? 1 : 0;
          const conf = named >= 1 && !errs ? 0.80 : (named >= 1 ? 0.55 : 0.30);
          return { tier: conf >= 0.55 ? 'hot' : 'abstain', confidence: conf, reason: `tree-sitter:${id}`, modelHint: 'claude-3-5-sonnet-latest' };
        } catch {
          return { tier: 'abstain', confidence: 0.0, reason: `tree-sitter:${id}-parse-error` };
        }
      }
      ```

    Step 3 — src/router/classifier/index.ts:

      // FORBIDDEN imports (enforced by tests/arch/no-network-in-classifier.test.ts):
      //   node:net | node:http(s) | undici | axios | ../ee/* | fetch()
      import type { ClassifierResult } from '../types.js';
      import { matchRegex } from './regex.js';
      import { lazyTreeSitter, initTreeSitter } from './tree-sitter.js';

      export function classify(prompt: string, threshold = 0.55): ClassifierResult {
        const r = matchRegex(prompt);
        if (r.confidence >= threshold) return r;
        const t = lazyTreeSitter(prompt);
        if (t.confidence >= threshold) return t;
        return { tier: 'abstain', confidence: Math.max(r.confidence, t.confidence), reason: 'low-confidence' };
      }
      export async function warm(): Promise<void> { await initTreeSitter(['typescript','python']); }

      import { describe, it, expect } from 'vitest';
      import { readdirSync, readFileSync } from 'node:fs';
      import { join } from 'node:path';

      const FORBIDDEN: Array<{ name: string; re: RegExp }> = [
        { name: 'node:net',    re: /from\s+['"]node:net['"]/ },
        { name: 'node:http',   re: /from\s+['"]node:https?['"]/ },
        { name: 'undici',      re: /from\s+['"]undici['"]/ },
        { name: 'axios',       re: /from\s+['"]axios['"]/ },
        { name: 'ee-import',   re: /from\s+['"](\.\.\/)+ee\// },
        { name: 'global-fetch',re: /\bfetch\s*\(/ },
      ];
      function* walk(dir: string): Generator<string> {
        for (const e of readdirSync(dir, { withFileTypes: true })) {
          const p = join(dir, e.name);
          if (e.isDirectory()) yield* walk(p);
          else if (/\.tsx?$/.test(e.name) && !/\.test\.tsx?$/.test(e.name)) yield p;
        }
      }
      describe('ROUTE-01: no network in hot-path classifier', () => {
        it('src/router/classifier/** must not import network APIs or call fetch()', () => {
          const offenders: string[] = [];
          for (const file of walk('src/router/classifier')) {
            const src = readFileSync(file, 'utf8');
            for (const f of FORBIDDEN) if (f.re.test(src)) offenders.push(`${file}: ${f.name}`);
          }
          expect(offenders).toEqual([]);
        });
      });

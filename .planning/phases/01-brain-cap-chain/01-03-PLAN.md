      ```typescript
      import { serve, type Server } from 'bun';
      export interface StubConfig {
        port?: number;
        routeModel?: (req: any) => any;        // returns RouteModelResponse or 500 if undefined
        coldRoute?: (req: any) => any;
        intercept?: (req: any) => any;
        posttool?: (req: any) => void;
        feedback?: (req: any) => void;
        touch?: (id: string) => void;
        health?: () => boolean;
        latencyMs?: number;                    // optional injected delay
      }
      export interface StubHandle { server: Server; port: number; stop(): Promise<void>; calls: Record<string, any[]> }
      export async function startStubEEServer(cfg: StubConfig = {}): Promise<StubHandle> {
        const calls: Record<string, any[]> = { intercept: [], posttool: [], routeModel: [], coldRoute: [], feedback: [], touch: [] };
        const port = cfg.port ?? 0;
        const server = serve({
          port,
          fetch: async (req) => {
            const url = new URL(req.url);
            if (cfg.latencyMs) await new Promise(r => setTimeout(r, cfg.latencyMs));
            if (url.pathname === '/health') return new Response(JSON.stringify({ ok: cfg.health ? cfg.health() : true }), { status: cfg.health && !cfg.health() ? 503 : 200 });
            const body = req.method === 'POST' ? await req.json().catch(() => ({})) : {};
            if (url.pathname === '/api/route-model') { calls.routeModel.push(body); const r = cfg.routeModel?.(body); return r ? Response.json(r) : new Response('no handler', { status: 500 }); }
            if (url.pathname === '/api/cold-route') { calls.coldRoute.push(body); const r = cfg.coldRoute?.(body); return r ? Response.json(r) : new Response('no handler', { status: 500 }); }
            if (url.pathname === '/api/intercept')  { calls.intercept.push(body);  const r = cfg.intercept?.(body)  ?? { decision: 'allow' }; return Response.json(r); }
            if (url.pathname === '/api/posttool')   { calls.posttool.push(body);   cfg.posttool?.(body); return new Response('ok'); }
            if (url.pathname === '/api/feedback')   { calls.feedback.push(body);   cfg.feedback?.(body); return new Response('ok'); }
            if (url.pathname.startsWith('/api/principle/touch')) { const id = url.searchParams.get('id') ?? body?.id; calls.touch.push(id); cfg.touch?.(id); return new Response('ok'); }
            return new Response('not found', { status: 404 });
          },
        });
        return { server, port: server.port, stop: async () => { server.stop(true); }, calls };
      }
      ```

    Step 1 — src/router/store.ts (small subscribable atom; Zustand-like, zero deps):
      ```typescript
      import type { RouteDecision, Tier } from './types.js';
      export interface RouterState {
        tier: Tier;                          // current display tier
        degraded: boolean;                   // health probe result
        lastDecision: RouteDecision | null;
        lastHealthCheckAtMs: number;
      }
      type Listener = (s: RouterState) => void;
      function makeStore() {
        let state: RouterState = { tier: 'hot', degraded: false, lastDecision: null, lastHealthCheckAtMs: 0 };
        const listeners = new Set<Listener>();
        return {
          getState: () => state,
          setState: (patch: Partial<RouterState>) => { state = { ...state, ...patch }; for (const l of listeners) l(state); },
          subscribe: (fn: Listener) => { listeners.add(fn); return () => listeners.delete(fn); },
        };
      }
      export const routerStore = makeStore();
      ```

      ```typescript
      import { getDefaultEEClient } from '../ee/intercept.js';
      import { routerStore } from './store.js';

      const PROBE_INTERVAL_MS = 30_000;
      const TTL_MS = 60_000;
      let handle: NodeJS.Timeout | null = null;
      let lastProbeMs = 0;
      let lastOk = true;

      async function probe(): Promise<void> {
        const ok = (await getDefaultEEClient().health()).ok;
        lastOk = ok;
        lastProbeMs = Date.now();
        routerStore.setState({ degraded: !ok, lastHealthCheckAtMs: lastProbeMs });
      }
      export async function startHealthProbe(): Promise<void> {
        if (handle) return;
        await probe(); // initial sync probe
        handle = setInterval(() => { void probe(); }, PROBE_INTERVAL_MS);
        if (typeof (handle as any).unref === 'function') (handle as any).unref();
      }
      export function stopHealthProbe(): void { if (handle) { clearInterval(handle); handle = null; } }
      export function getHealthStatus(): { ok: boolean; staleMs: number } {
        return { ok: lastOk, staleMs: Date.now() - lastProbeMs };
      }
      // helpers used internally + by tests:
      export const __testing = { TTL_MS, PROBE_INTERVAL_MS };
      ```

    Step 5 — src/router/decide.ts:

      ```typescript
      import { classify } from './classifier/index.js';
      import { callWarmRoute } from './warm.js';
      import { callColdRoute } from './cold.js';
      import { routerStore } from './store.js';
      import type { RouteDecision } from './types.js';

      export interface DecideOpts { tenantId: string; cwd: string; threshold?: number; signal?: AbortSignal; defaultModel: string; defaultProvider: string }

      export async function decide(prompt: string, opts: DecideOpts): Promise<RouteDecision> {
        const c = classify(prompt, opts.threshold ?? 0.55);
        if (c.tier === 'hot') {
          const d: RouteDecision = { tier: 'hot', model: c.modelHint ?? opts.defaultModel, provider: opts.defaultProvider, reason: c.reason, confidence: c.confidence };
          routerStore.setState({ tier: 'hot', lastDecision: d });
          return d;
        }
        const w = await callWarmRoute(prompt, opts);
        if (w) { routerStore.setState({ tier: w.tier, lastDecision: w }); return w; }
        const cd = await callColdRoute(prompt, opts);
        if (cd) { routerStore.setState({ tier: 'cold', lastDecision: cd }); return cd; }
        // Final fallback when EE entirely unreachable
        const fallback: RouteDecision = { tier: routerStore.getState().degraded ? 'degraded' : 'hot', model: opts.defaultModel, provider: opts.defaultProvider, reason: 'fallback:ee-unreachable' };
        routerStore.setState({ lastDecision: fallback });
        return fallback;
      }
      ```
